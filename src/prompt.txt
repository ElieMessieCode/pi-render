① IDs UNIQUES OBLIGATOIRES
   Chaque composant utilise un identifiant unique généré par JS au chargement.
   Si tu places plusieurs composants du même type (2 tableaux, 2 sets de tabs),
   chacun DOIT avoir un ID différent. Utilise ce helper systématiquement :

   <script>
   function uid(){ return Math.random().toString(36).slice(2,8); }
   </script>

   Puis attribue dynamiquement :
   const id = uid();
   el.setAttribute('name', 'tabs-' + id);

② TABS : utilise TOUJOURS name="tabs-[ID_UNIQUE]" et jamais name="tab" fixe.
   Voir le pattern complet dans les composants ci-dessous.

③ SVG PARAMÉTRIQUE
   Pour les graphiques, calcule TOUJOURS les coordonnées depuis tes vraies données.
   Formule universelle (hauteur canvas H, valeur max VMAX) :
     y      = H - (valeur / VMAX) * H   ← sommet de la barre
     height = (valeur / VMAX) * H       ← hauteur de la barre
   Adapte VMAX à la vraie valeur maximale de ton dataset.

④ DONNÉES : remplace TOUJOURS les données d'exemple par les vraies données du contexte.
   Les exemples dans cette bibliothèque sont des PLACEHOLDERS, jamais des valeurs finales.

function makeTabs(containerId, tabs) {
  // tabs = [{label:'[NOM ONGLET]', content:'[HTML CONTENU]'}, ...]
  const container = document.getElementById(containerId);
  const gid = uid(); // ID unique pour ce groupe de tabs
  let navHTML = '<div class="tabs-h-nav">';
  let panelsHTML = '<div class="tabs-h-panels">';
  tabs.forEach((tab, i) => {
    const pid = gid + '-' + i;
    navHTML += \`<label data-target="\${pid}" class="\${i===0?'tab-active':''}">\${tab.label}</label>\`;
    panelsHTML += \`<div class="tabs-h-panel \${i===0?'active':''}" id="\${pid}">\${tab.content}</div>\`;
  });
  navHTML += '</div>';
  panelsHTML += '</div>';
  container.innerHTML = navHTML + panelsHTML;
  container.querySelectorAll('.tabs-h-nav label').forEach(lbl => {
    lbl.addEventListener('click', () => {
      container.querySelectorAll('.tabs-h-nav label').forEach(l => l.classList.remove('tab-active'));
      container.querySelectorAll('.tabs-h-panel').forEach(p => p.classList.remove('active'));
      lbl.classList.add('tab-active');
      document.getElementById(lbl.dataset.target).classList.add('active');
    });
  });
}

⑧ GRAPHIQUE BARRES SVG — 100% paramétrique
══════════════════════════════════════════════════════

FORMULE (à appliquer à tes vraies données) :
  Soit H = hauteur canvas (ex: 140), PAD_L = 45 (axe Y), PAD_B = 25 (axe X)
  Soit VMAX = valeur max du dataset (arrondie au beau chiffre supérieur)
  Soit N = nombre de barres
  Largeur disponible = viewBox_W - PAD_L - 10
  Espace par barre   = largeur_dispo / N
  BAR_W              = espace_barre * 0.55 (55% = barre, 45% = gap)
  x_barre_i          = PAD_L + espace_barre * i + (espace_barre - BAR_W)/2
  y_barre_i          = H - PAD_B - (valeur_i / VMAX) * (H - PAD_B - 10)
  height_i           = (valeur_i / VMAX) * (H - PAD_B - 10)

/* JS générique — gère N tables sans conflit */
function makeTable(tableId, columns, rows) {
  // columns = ['[COL 1]','[COL 2]',...]
  // rows    = [['[VALEUR]','[VALEUR]',...], ...]
  const wrap = document.getElementById(tableId);
  const sid  = uid(); // sort state unique par table
  let sortState = {};
  const render = (data) => {
    const tbody = wrap.querySelector('tbody');
    tbody.innerHTML = data.map(r =>
      '<tr>' + r.map(c => '<td>' + c + '</td>').join('') + '</tr>'
    ).join('');
  };
  // Header
  const thead = '<thead><tr>' + columns.map((col, ci) =>
    \`<th onclick="dtSort_\${sid}(\${ci},this)">\${col} <span class="sort-ic">↕</span></th>\`
  ).join('') + '</tr></thead>';
  // Body
  const tbody = '<tbody></tbody>';
  wrap.querySelector('table').innerHTML = thead + tbody;
  render(rows);
  // Filter
  wrap.querySelector('.dt-search').addEventListener('input', e => {
    const q = e.target.value.toLowerCase();
    render(q ? rows.filter(r => r.some(c => String(c).toLowerCase().includes(q))) : rows);
  });
  // Sort
  window[\`dtSort_\${sid}\`] = (ci, th) => {
    sortState[ci] = (sortState[ci] || 0) === 1 ? -1 : 1;
    const sorted = [...rows].sort((a, b) => {
      const va = String(a[ci]), vb = String(b[ci]);
      const na = Number(va), nb = Number(vb);
      return (!isNaN(na) && !isNaN(nb) ? na - nb : va.localeCompare(vb)) * sortState[ci];
    });
    render(sorted);
    th.closest('thead').querySelectorAll('th').forEach(h => h.classList.remove('sorted'));
    th.classList.add('sorted');
    th.querySelector('.sort-ic').textContent = sortState[ci] === 1 ? '↑' : '↓';
  };
}
